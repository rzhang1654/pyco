149  const chrono::time_point<__clock_t, _Dur>& __atime)
150  {
151  auto __s = chrono::time_point_cast<chrono::seconds>(__atime);
152  auto __ns = chrono::duration_cast<chrono::nanoseconds>(__atime - __s);

154  __gthread_time_t __ts =
